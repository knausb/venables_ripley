10.1 Linear models

library(nlme)
library(MASS) # For the petrol data set.
data(petrol)
library(lattice) # xyplot
xyplot(Y ~ EP | No, data = petrol,
       xlab = "ASTM end point (deg. F)",
       ylab = "Yield as a percent of crude",
       panel = function(x, y){
         panel.grid()
         m <- sort.list(x)
         panel.xyplot(x[m], y[m], type = "b", cex = 0.5)
       })
